create table administrador(
	matricula varchar(30) primary key,
	nome varchar(30),
	senha varchar(30),
	categoria varchar(30),
	status integer
);

create table empresa( nome varchar(30),
	cnpj varchar(30) primary key,
	email varchar(30),
	r_social varchar(30),
	responsavel varchar(30), 
	cadastro varchar(30),
	cargo varchar(30),
	cep varchar(30),
	cidade varchar(30),
	estado varchar(30),
	endereco varchar(30),
	numero varchar(30),
	bairro varchar(30),
	telefone varchar(30),
	senha varchar(30),
	categoria varchar(30),
	status integer
);

create table supervisor( 
	matricula varchar(30) primary key,
	nome varchar(30),
	senha varchar(30),
	categoria varchar(30),
	status integer
);

create table usuario(nome varchar(30),
	matricula varchar(30) primary key,
	curso varchar(30),
	fone varchar(30), 
	email varchar(30),
	senha varchar(30),
	categoria varchar(30),
	status integer
);

create table curriculo(
	nome varchar(30),
	formacao varchar(30),
	experiencia varchar(30),
	matAluno varchar(30) primary key,
	status varchar(30),
	foreign key(matAluno) references usuario(matricula)
);
create table empresaCRL(
	numero varchar(30) primary key,
	nome varchar(30),
	obrigatorio varchar(30),
	desejavel varchar(30),
	descricao varchar(30),
	cnpj varchar(30),
	status integer,
	foreign key(cnpj) references empresa(cnpj)
);
